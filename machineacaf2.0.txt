#!/bin/bash

# Présentation du menu

menu () {

choice=0
choice_sortie=0

while [ $choice_sortie != 1 ]
do

echo "
=============
Menu : 
1 Expresso
2 Café Long
3 Thé
4 Chocolat
5 MenuTech
6 MenuStats
7 Sortie
============="

read choice

# Renseigner type de paiement
# Renseigner montant en fonction du choix
# Prise en compte de la présence d'un gobelet, si oui, sa taille
# Renseigner quantité de sucre si > 2 touillette

case $choice in
1)
	echo "Choix Expresso"
	taille_gobelet=petite;;

2)
	echo "Choix Café Long"
	taille_gobelet=grande;;

3)
	echo "Choix Thé"
	taille_gobelet=grande;;

4)
	echo "Choix Chocolat"
	taille_gobelet=grande;;
5)
	echo "Choix MenuTech";;

6)
	echo "Choix MenuStats";;

7)
	echo "Sortie du menu."
	choice_sortie=1;;

*)
 	echo "Le choix n'est pas bon."
  	menu;;
esac

# Choix gobelet oui ou non (à mettre dans chaque choix de boisson)

read -p "Besoin d'un gobelet ? (oui(1) / non(2))" gobelet

if [ $gobelet = 1 ]
then
	echo "Un gobelet sera servi"
	echo "Service de la boisson dans un Gobelet $taille_gobelet taille"
else
	echo "Pas de gobelet servi"
	echo "Service de la boisson dans la tasse"
fi


#$quantitesucre choisit, $monnaie rendue"

done

}
menu
